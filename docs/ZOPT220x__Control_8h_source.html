297 // Sets the mode of the sensor
298 // Mode 1: Standby (power on default)
299 // Mode 2: Ambient light sensing w/ compensation
300 // Mode 3: UVB sensing w/ compensation
301 // Mode 4: ALS/UVBS_Raw
302 void setMode(byte mode) {
303 // Read, set, write
304 byte value = zoptReadRegister(ZOPT220x_MAIN_CTRL); // Read
305
306 if (mode == 0)
307 value &= ~(1 << 1); // Not a valid mode. Standby. Clear LS_EN bit
308 else if (mode == 1)
309 value &= ~(1 << 1); // Standby. Clear LS_EN bit
310 else if (mode == 2) // ALS
311 {
312 value |= (1 << 1); // Set LS_EN bit
313 value &= ~(1 << 2); // Clear RawMode_SEL bit
314 value &= ~(1 << 3); // Clear LS_Mode bit for ALS
315 } else if (mode == 3) // UVB
316 {
317 value |= (1 << 1); // Set LS_EN bit
318 value &= ~(1 << 2); // Clear RawMode_SEL bit
319 value |= (1 << 3); // Set LS_Mode bit for UVB
320 } else if (mode == 4) // RawMode
321 {
322 // Send magic twiddly bytes
323 zoptWriteRegister(ZOPT220x_SPECIAL_MODE_2,
324 0xB5); // Write first special byte
325 zoptWriteRegister(ZOPT220x_SPECIAL_MODE_1,
326 0xDF); // Write second special byte
327 zoptWriteRegister(ZOPT220x_SPECIAL_MODE_2,
328 0x04); // Write third special byte
329
330 value |= (1 << 1); // Set LS_EN bit
331 value |= (1 << 2); // Set RawMode_SEL bit
332 value &= ~(1 << 3); // Clear LS_Mode bit
333 }
334
335 zoptWriteRegister(ZOPT220x_MAIN_CTRL, value); // Write
336 }

363 // The ALS and UVB data come in three byte pack: low/mid/high
364 // Read all three and store in long
365 // Returns -1 if times out
366 // Max size is 20 bits so signed long is ok
367 long zoptRead3Registers(byte addr) {
368 Wire.beginTransmission(zoptAddress);
369 Wire.write(addr);
370 Wire.endTransmission();
371
372 Wire.requestFrom(zoptAddress, 3);
373
374 // Non-blocking wait for 3 bytes
375 byte counter = 0;
376 while (Wire.available() < 3) {
377 delay(1);
378 if (counter++ > 25) return (-1); // Error
379 }
380
381 byte lowValue = Wire.read();
382 byte midValue = Wire.read();
383 byte highValue = Wire.read();
384
385 long value = (long)highValue << 16 | (long)midValue << 8 | lowValue;
386
387 return (value);
388 }
