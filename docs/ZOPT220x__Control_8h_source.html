300  // Read, set, write

302 
303  if (mode == 0)
304  value &= ~(1 << 1); // Not a valid mode. Standby. Clear LS_EN bit
305  else if (mode == 1)
306  value &= ~(1 << 1); // Standby. Clear LS_EN bit
307  else if (mode == 2) // ALS

364 long zoptRead3Registers(byte addr) {
365  Wire.beginTransmission(zoptAddress);
366  Wire.write(addr);
367  Wire.endTransmission();
368 
369  Wire.requestFrom(zoptAddress, 3);
370 
371  // Non-blocking wait for 3 bytes
372  byte counter = 0;
373  while (Wire.available() < 3) {
374  delay(1);
375  if (counter++ > 25) return (-1); // Error
376  }
377 
378  byte lowValue = Wire.read();
379  byte midValue = Wire.read();
380  byte highValue = Wire.read();
381 
382  long value = (long)highValue << 16 | (long)midValue << 8 | lowValue;
383 
384  return (value);
385 }
